\قسمت{تراکنش های کانال پرداخت}
\subsection{ایجاد کانال}
همانطور که پیش از این گفته شد، دو کاربر برای ایجاد کانال پرداخت باید یک تراکنش درون بلاکچینی "ایجاد کانال" بسازند. شکل زیر نمونه ای از یک تراکنش ایجاد کانال است که در آن \کد{A} یکی از \کد{UTXO} هایش به ارزش $2$ بیتکوین و B یکی از \کد{UTXO} هایش به ارزش $4$ بیتکوین را در کانال ذخیره میکنند. خروجی این تراکنش یک حساب دو کاربره با موجودی 6 است. 

\شروع{شکل}[hb]
\centerimg{channelCreation.png}{8cm}
\شرح{مثالی از یک تراکنش ایجاد کانال پرداخت. این تراکنش درون بلاکچینی است یعنی روی بلاکچین بیتکوین فرستاده میشود.}
\برچسب{شکل:تراکنش ایجاد کانال}
\پایان{شکل}

همزمان با تراکنش درون بلاکچینی \کد{Tx1}، \کد{A}  و  \کد{B} مشترکا یک تراکنش برون بلاکچینی (\کد{Tx2}) هم ساخته و هر دو آن امضا میکنند اما آن را روی بلاکچین نمیفرستند، این تراکنش تنها در حافظه محلی \کد{A}  و  \کد{B} ذخیره میشود. 

\شروع{شکل}[hb]
\centerimg{state0.png}{6cm}
\شرح{تراکنش حالت 0 کانال پرداخت. این کانال برون بلاکچینی است یعنی توسط \کد{A}  و  \کد{B} مشترکا ساخته و امضا شده و ذخیره می شود ولی تا زمان بسته شدن کانال روی بلاکچین قرار نمیگیرد.}
\برچسب{شکل:تراکنش حالت}
\پایان{شکل}


\کد{Tx2} تک خروجی تراکنش \کد{Tx1} را خرج میکند و خروجی هایش، پول موجود در کانال را به همان نسبت اولیه $2-4$ بین \کد{A}  و  \کد{B} تقسیم میکند. برای استفاده از  خروجی \کد{Tx2} حداقل 30 روز از نهایی شدن آن در بلاکچین باید گذشته باشد. \کد{Tx2} در واقع توزیع پول در کانال در لحظه ایجاد آن را نشان میدهد. در صورتی که هیچ تراکنشی بین \کد{A}  و  \کد{B}  انجام نگیرد و پس از مدتی یکی از \کد{A}  یا  \کد{B} تصمیم بگیرد کانال را ببندد، آن فرد میتواند تراکنش \کد{Tx2} را روی بلاکچین قرار دهد. چون \کد{Tx2}  پیش از این توسط هر دو \کد{A}  و  \کد{B} امضا شده است پس تراکنش معتبری است و پس از 30 روز از ارسال آن  \کد{A}  و  \کد{B}  هر کدام میتوانند سهم خود را از کانال پرداخت بگیرند و کانال بسته میشود.
